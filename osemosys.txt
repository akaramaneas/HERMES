param YearWeight {y in YEAR};

param DiscountFactor{r in REGION, y in YEAR} :=
	if y = min{yy in YEAR} min(yy) then (1 + DiscountRate[r])^(y - min{yy in YEAR} min(yy) + 1)
	else if y = max{yy in YEAR} max(yy) then (1 + DiscountRate[r])^(y - min{yy in YEAR} min(yy))
	else (1 + DiscountRate[r])^(y -min{yy in YEAR} min(yy) + 2.5);	
param DiscountFactorMid{r in REGION, y in YEAR} :=
	if y = min{yy in YEAR} min(yy) then (1 + DiscountRate[r])^(y - min{yy in YEAR} min(yy) + 1)
	else if y = max{yy in YEAR} max(yy) then (1 + DiscountRate[r])^(y - min{yy in YEAR} min(yy))
	else (1 + DiscountRate[r])^(y -min{yy in YEAR} min(yy) + 2.5);

param OperationalLife{r in REGION, t in TECHNOLOGY};

param CapitalRecoveryFactor{r in REGION, t in TECHNOLOGY} := DiscountRateIdv[r,t] / (1 - (1 / ( (1 +DiscountRateIdv[r,t])^(OperationalLife[r,t]))));
#	(1 - (1 + DiscountRateIdv[r,t])^(-1))/(1 - (1 + DiscountRateIdv[r,t])^(-(OperationalLife[r,t])));

param OperationalLifeStorage{r in REGION, s in STORAGE};

param CapitalRecoveryFactorStorage{r in REGION, s in STORAGE} := DiscountRateStorage[r,s] / (1 - (1 / ( (1 +DiscountRateStorage[r,s])^(OperationalLifeStorage[r,s]))));

param PvAnnuity{r in REGION, t in TECHNOLOGY} := 1;
#	(1 - (1 + DiscountRate[r])^(-(OperationalLife[r,t]))) * (1 + DiscountRate[r]) / DiscountRate[r];

#param DiscountRateStorage{r in REGION, s in STORAGE}; #defined above
param DiscountFactorStorage{r in REGION, s in STORAGE, y in YEAR} :=
	(1 + DiscountRateStorage[r,s]) ^ (y - min{yy in YEAR} min(yy) + 0.0);

set MODExTECHNOLOGYperFUELout{r in REGION, f in FUEL} within MODE_OF_OPERATION cross TECHNOLOGY :=
{m in MODE_OF_OPERATION, t in TECHNOLOGY : exists{y in YEAR} OutputActivityRatio[r,t,f,m,y] <> 0};

set MODExTECHNOLOGYperFUELin{r in REGION,f in FUEL} within MODE_OF_OPERATION cross TECHNOLOGY :=
{m in MODE_OF_OPERATION, t in TECHNOLOGY : exists{y in YEAR} InputActivityRatio[r,t,f,m,y] <> 0};

set MODExTECHNOLOGYperSTORAGEto{r in REGION, s in STORAGE} within MODE_OF_OPERATION cross TECHNOLOGY :=
{m in MODE_OF_OPERATION, t in TECHNOLOGY : TechnologyToStorage[r,t,s,m] > 0};

set MODExTECHNOLOGYperSTORAGEfrom{r in REGION, s in STORAGE} within MODE_OF_OPERATION cross TECHNOLOGY :=
{m in MODE_OF_OPERATION, t in TECHNOLOGY : TechnologyFromStorage[r,t,s,m] > 0};

set TIMESLICEofSEASON{ls in SEASON} within TIMESLICE := {l in TIMESLICE : Conversionls[l,ls] = 1};
set TIMESLICEofDAYTYPE{ld in DAYTYPE} within TIMESLICE := {l in TIMESLICE : Conversionld[l,ld] = 1};
set TIMESLICEofDAILYTIMEBRACKET{lh in DAILYTIMEBRACKET} within TIMESLICE := {l in TIMESLICE : Conversionlh[l,lh] = 1};
set TIMESLICEofSDB{ls in SEASON, ld in DAYTYPE, lh in DAILYTIMEBRACKET} within TIMESLICE := TIMESLICEofSEASON[ls] inter TIMESLICEofDAYTYPE[ld] inter TIMESLICEofDAILYTIMEBRACKET[lh];

set MODExTECHNOLOGYperEMISSION{r in REGION,e in EMISSION} within MODE_OF_OPERATION cross TECHNOLOGY :=
{m in MODE_OF_OPERATION, t in TECHNOLOGY : exists{y in YEAR} EmissionActivityRatio[r,t,e,m,y] <> 0};

set MODEperTECHNOLOGY{r in REGION, t in TECHNOLOGY} within MODE_OF_OPERATION :=
{m in MODE_OF_OPERATION : (exists {f in FUEL} (m, t) in MODExTECHNOLOGYperFUELout[r,f] union MODExTECHNOLOGYperFUELin[r,f])
 or (exists {s in STORAGE} (m, t) in MODExTECHNOLOGYperSTORAGEto[r,s]
 union MODExTECHNOLOGYperSTORAGEfrom[r,s])
 or (exists {e in EMISSION} (m, t) in MODExTECHNOLOGYperEMISSION[r,e])};


set REGION_FUEL_VALID within REGION cross FUEL :=
    {r in REGION, f in FUEL :
        card(MODExTECHNOLOGYperFUELout[r,f]) > 0
        or card(MODExTECHNOLOGYperFUELin[r,f]) > 0};

set REGION_TECH_VALID within REGION cross TECHNOLOGY :=
    { r in REGION, t in TECHNOLOGY :
        exists{f in FUEL, m in MODE_OF_OPERATION, y in YEAR}
            ( InputActivityRatio[r,t,f,m,y] <> 0
              or OutputActivityRatio[r,t,f,m,y] <> 0 )};

set REGION_STOR_VALID within REGION cross STORAGE :=
    { r in REGION, s in STORAGE :
        exists{t in TECHNOLOGY, m in MODE_OF_OPERATION}
            ( TechnologyToStorage[r,t,s,m] > 0
              or TechnologyFromStorage[r,t,s,m] > 0 )};

set REGION_INTER_VALID within REGION cross INTERCONNECTION :=
    { r in REGION, i in INTERCONNECTION :
        exists{y in YEAR} EfficiencyInterconnection[r,i,y]>0};


set REGION_TECH_FUEL_VALID within REGION cross TECHNOLOGY cross FUEL:=
    { r in REGION, t in TECHNOLOGY, f in FUEL :
        exists{m in MODE_OF_OPERATION, y in YEAR}

              or OutputActivityRatio[r,t,f,m,y] <> 0 )};

            ( ResidualCapacityInterconnection[r,i,f,y] <> 0)};

table AccumulatedNewCapacity
    {r in REGION, t in TECHNOLOGY, y in YEAR:
        sum{yy in YEAR: y-yy < OperationalLife[r,t] && y-yy>=0}
        NewCapacity[r,t,yy] > 0}
    OUT "CSV"
    ResultsPath & "/AccumulatedNewCapacity.csv" :
    r~REGION, t~TECHNOLOGY, y~YEAR,
    sum{yy in YEAR: y-yy < OperationalLife[r,t] && y-yy>=0}
        NewCapacity[r,t,yy] ~VALUE;

table AnnualFixedOperatingCost
    {r in REGION, t in TECHNOLOGY, y in YEAR:
        ((sum{yy in YEAR: y-yy < OperationalLife[r,t] && y-yy>=0}
            NewCapacity[r,t,yy]) + ResidualCapacity[r,t,y])
        > 0 && FixedCost[r,t,y] > 0
        }
    OUT "CSV"
    ResultsPath & "/AnnualFixedOperatingCost.csv" :
    r~REGION, t~TECHNOLOGY, y~YEAR,
    FixedCost[r,t,y] *
    ((sum{yy in YEAR: y-yy < OperationalLife[r,t] && y-yy>=0}
        NewCapacity[r,t,yy]) + ResidualCapacity[r,t,y]) ~VALUE;

table TotalDiscountedCostResults
	{r in REGION, y in YEAR}
	OUT "CSV"
	ResultsPath & "/TotalDiscountedCost.csv":
	r~REGION, y~YEAR,
	sum{t in TECHNOLOGY}
        ((((sum{yy in YEAR: y-yy < OperationalLife[r,t] && y-yy>=0}
            NewCapacity[r,t,yy]) + ResidualCapacity[r,t,y]) * FixedCost[r,t,y]
          + sum{l in TIMESLICE, m in MODEperTECHNOLOGY[r,t]}
                RateOfActivity[r,l,t,m,y] * YearSplit[l,y] * VariableCost[r,t,m,y]) / (DiscountFactorMid[r,y])
          + CapitalCost[r,t,y] * NewCapacity[r,t,y] * CapitalRecoveryFactor[r,t] * PvAnnuity[r,t]/ (DiscountFactor[r,y])
          + DiscountedTechnologyEmissionsPenalty[r,t,y]
          - DiscountedSalvageValue[r,t,y])
          + sum{s in STORAGE}
                (CapitalCostStorage[r,s,y] * NewStorageCapacity[r,s,y] / (DiscountFactorStorage[r,s,y])
                 - CapitalCostStorage[r,s,y] * NewStorageCapacity[r,s,y] / (DiscountFactorStorage[r,s,y])
          )~VALUE;
